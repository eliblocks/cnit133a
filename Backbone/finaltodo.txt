/* rendering all the list of todo items calling TodoView for each one */
    app.AppView = Backbone.View.extend({
      el: '#todoapp',
      initialize: function () {
        this.input = this.$('#newtodo');
        app.todoList.on('add', this.addOne, this);
        app.todoList.on('reset', this.addAll, this);
        app.todoList.fetch(); 
      },
      events: {
        'keypress #newtodo': 'createTodoOnEnter'
      },
      createTodoOnEnter: function(e){
        if ( e.which !== 13 || !this.input.val().trim() ) { // The ENTER key is = 13
          return;
        }
        app.todoList.create(this.newAttributes());
        this.input.val('');  // cleaning input box
      },
      addOne: function(todo){
        var view = new app.TodoView({model: todo});
        $('#todolist').append(view.render().el);
      },
      addAll: function(){
        this.$('#todolist').html(''); // cleaning todo list
        app.todoList.each(this.addOne, this);
      },
      newAttributes: function(){
        return {
          title: this.input.val().trim(),
          completed: false
        }
      }
    });

    // Initializers

    app.appView = new app.AppView();
